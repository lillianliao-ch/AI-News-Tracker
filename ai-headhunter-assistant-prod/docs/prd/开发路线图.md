```python
# services/resume_parser.py
import pdfplumber

class ResumeParser:
    def parse_pdf(self, pdf_file):
        """解析 PDF 简历"""
        text = ""
        with pdfplumber.open(pdf_file) as pdf:
            for page in pdf.pages:
                text += page.extract_text()
        
        return {
            "raw_text": text,
            "page_count": len(pdf.pages),
            "parser_type": "pdf"
        }
```

**评估算法**：
```python
def calculate_match_score(candidate, jd_config):
    """
    计算综合匹配度
    """
    weights = jd_config["evaluation_weights"]
    
    skill_score = evaluate_skill_match(candidate, jd_config)
    experience_score = evaluate_experience_match(candidate, jd_config)
    education_score = evaluate_education_match(candidate, jd_config)
    stability_score = evaluate_stability(candidate)
    
    total_score = (
        skill_score * weights["skill"] +
        experience_score * weights["experience"] +
        education_score * weights["education"] +
        stability_score * weights["stability"]
    )
    
    return total_score
```

```python
# services/feishu_client.py
from lark_oapi import Client

class FeishuClient:
    def __init__(self):
        self.client = Client.builder() \
            .app_id(FEISHU_APP_ID) \
            .app_secret(FEISHU_APP_SECRET) \
            .build()
    
    def create_candidate_record(self, candidate_data):
        """创建候选人记录"""
        record = {
            "fields": {
                "候选人ID": candidate_data["candidate_id"],
                "姓名": candidate_data["name"],
                "匹配度": candidate_data["match_score"],
                # ...
            }
        }
        
        response = self.client.bitable.app_table_record.create(
            app_token=FEISHU_APP_TOKEN,
            table_id=FEISHU_TABLE_ID,
            record=record
        )
        
        return response.data.record_id
```

```python
# tasks/process_candidate.py
from celery import Celery

app = Celery('tasks', broker='redis://localhost:6379/1')

@app.task(bind=True, max_retries=3)
def process_candidate_task(self, candidate_url, jd_config):
    try:
        # 1. 采集简历
        resume = download_resume(candidate_url)
        
        # 2. AI 评估
        evaluation = ai_evaluate(resume, jd_config)
        
        # 3. 入库飞书
        feishu_client.create_record(evaluation)
        
        # 4. 打招呼（如果需要）
        if evaluation['recommend_level'] == '推荐':
            auto_greeting(candidate_url, evaluation['greeting'])
        
        return {'status': 'success'}
    
    except Exception as e:
        # 重试
        raise self.retry(exc=e, countdown=60)
```
