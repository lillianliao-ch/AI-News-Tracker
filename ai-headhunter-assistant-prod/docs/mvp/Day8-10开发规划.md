```python
# services/feishu_client.py

async def upload_image(self, image_base64: str, filename: str) -> str:
    """上传图片到飞书云盘
    
    Args:
        image_base64: Base64 编码的图片
        filename: 文件名
        
    Returns:
        file_token: 飞书文件 Token
    """
    # 1. 解码 Base64
    image_data = base64.b64decode(image_base64)
    
    # 2. 调用飞书上传接口
    response = await self.client.upload_file(
        parent_type="bitable_image",
        parent_node="",
        file_data=image_data,
        file_name=filename
    )
    
    # 3. 返回 file_token
    return response.data.file_token
```

```javascript
// content-full.js

async function sendGreeting(candidate, greetingText) {
  try {
    // 1. 打开候选人详情
    await openDetail(candidate);
    
    // 2. 找到"打招呼"按钮（多个选择器备份）
    const greetBtn = document.querySelector('.greet-btn') 
      || document.querySelector('[data-action="greet"]')
      || document.querySelector('button:contains("打招呼")');
    
    if (!greetBtn) {
      console.log('❌ 未找到打招呼按钮');
      return { success: false, reason: '按钮未找到' };
    }
    
    // 3. 点击按钮
    greetBtn.click();
    await Utils.sleep(2000);
    
    // 4. 找到输入框
    const textarea = document.querySelector('.chat-input')
      || document.querySelector('textarea[placeholder*="打招呼"]');
    
    if (!textarea) {
      console.log('❌ 未找到输入框');
      return { success: false, reason: '输入框未找到' };
    }
    
    // 5. 模拟人工输入（逐字输入，随机延迟）
    await typeText(textarea, greetingText);
    
    // 6. 等待 2-5 秒（模拟人工思考）
    await Utils.sleep(Utils.random(2000, 5000));
    
    // 7. 找到发送按钮
    const sendBtn = document.querySelector('.send-btn')
      || document.querySelector('[data-action="send"]');
    
    if (!sendBtn) {
      console.log('❌ 未找到发送按钮');
      return { success: false, reason: '发送按钮未找到' };
    }
    
    // 8. 点击发送
    sendBtn.click();
    
    console.log(`✅ 成功向 ${candidate.name} 发送招呼`);
    return { success: true, sentAt: new Date().toISOString() };
    
  } catch (error) {
    console.error('❌ 发送招呼失败', error);
    return { success: false, reason: error.message };
  }
}

// 模拟人工输入
async function typeText(textarea, text) {
  textarea.value = '';
  textarea.focus();
  
  for (let char of text) {
    textarea.value += char;
    // 触发 input 事件
    textarea.dispatchEvent(new Event('input', { bubbles: true }));
    // 随机延迟 50-150ms
    await Utils.sleep(Utils.random(50, 150));
  }
}
```

```javascript
// 限流配置
const RATE_LIMIT = {
  maxPerDay: 50,           // 每日最多 50 条
  minInterval: 60000,      // 最小间隔 60 秒
  maxInterval: 180000,     // 最大间隔 180 秒
  workHourStart: 9,        // 工作时间开始
  workHourEnd: 21,         // 工作时间结束
  maxRetry: 3              // 最多重试 3 次
};

// 检查是否可以发送
function canSendGreeting() {
  const now = new Date();
  const hour = now.getHours();
  
  // 1. 检查工作时间
  if (hour < RATE_LIMIT.workHourStart || hour >= RATE_LIMIT.workHourEnd) {
    console.log('⏰ 非工作时间，暂停发送');
    return false;
  }
  
  // 2. 检查每日上限
  const todaySent = getTodaySentCount();
  if (todaySent >= RATE_LIMIT.maxPerDay) {
    console.log('⚠️ 今日已达上限，停止发送');
    return false;
  }
  
  // 3. 检查发送间隔
  const lastSentTime = getLastSentTime();
  const timeSinceLastSent = now - lastSentTime;
  if (timeSinceLastSent < RATE_LIMIT.minInterval) {
    console.log('⏳ 发送间隔太短，等待中...');
    return false;
  }
  
  return true;
}
```

## 💡 开发建议

### 1. 优先级管理

**Day 8 是关键**:
- 飞书集成是 MVP 的核心价值
- 必须在 Day 8 完成并测试通过
- 如果 Day 8 延期，可以压缩 Day 9/10

**Day 9 可灵活调整**:
- 自动打招呼有账号封禁风险
- 可以先做"半自动"（生成话术，手动发送）
- 如果时间紧张，可以放到 v1.1

**Day 10 不可省略**:
- 批量测试是质量保证
- 文档是交付必需品
- 没有测试和文档，系统无法交付

### 2. 风险控制

**飞书 API 风险**:
- 提前注册应用（可能需要审核）
- 提前测试 API 调用（避免配额问题）
- 准备 Plan B（如果飞书 API 失败，至少保证本地导出）

**打招呼风险**:
- Boss 直聘可能检测到自动化
- 建议先手动测试 10 条
- 发现异常立即停止
- 准备备用账号

**时间风险**:
- 每天设置检查点（中午、晚上）
- 如果某个任务超时，立即调整计划
- 优先保证 P0 任务完成

### 3. 测试策略

**渐进式测试**:
```
1 人测试 → 5 人测试 → 20 人测试 → 50 人测试
```

**每个阶段都要验证**:
- 功能正确性
- 性能指标
- 异常处理
- 数据完整性

**快速失败**:
- 发现问题立即停止
- 修复后再继续
- 不要在有 bug 的情况下继续
